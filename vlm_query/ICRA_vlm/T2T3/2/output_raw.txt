# The task involves 3 stages:
# 1. Robot 1 picks up the tweezers and Robot 2 picks up the scissors.
# 2. Robot 1 holds tweezers above 10 cm and Robot 2 holds scissors above 10 cm.
# 3. Robot 1 releases the tweezers on the desk and Robot 2 releases the scissors on the desk.

num_stages = 3

### stage 1 sub-goal constraints
def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector is firmly gripping the keypoint of the tweezers."""
    tweezers_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - tweezers_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Ensure Robot 2's end-effector is firmly gripping the keypoint of the scissors."""
    scissors_keypoint = keypoints[3]
    cost = np.linalg.norm(end_effector_2 - scissors_keypoint)
    return cost

    tweezers_keypoint = keypoints[1]
    offsetted_point = tweezers_keypoint + np.array([0, 0, 0.1])
    cost = np.linalg.norm(end_effector_1 - offsetted_point)
    return cost

def stage2_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Ensure Robot 2's end-effector holds the scissors 10 cm above the desk."""
    scissors_keypoint = keypoints[3]
    offsetted_point = scissors_keypoint + np.array([0, 0, 0.1])
    cost = np.linalg.norm(end_effector_2 - offsetted_point)
    return cost

### stage 3 sub-goal constraints
def stage3_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector releases the tweezers back on the desk."""
    tweezers_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - tweezers_keypoint)
    return cost

def stage3_subgoal_constraint_robot2(end_effector_2, keypoints):
