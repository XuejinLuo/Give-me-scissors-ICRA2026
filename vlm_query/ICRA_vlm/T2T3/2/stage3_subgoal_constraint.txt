def stage3_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector releases the tweezers back on the desk."""
    tweezers_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - tweezers_keypoint)
    return cost
