def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector is firmly gripping the keypoint of the tweezers."""
    tweezers_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - tweezers_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Ensure Robot 2's end-effector is firmly gripping the keypoint of the scissors."""
    scissors_keypoint = keypoints[3]
    cost = np.linalg.norm(end_effector_2 - scissors_keypoint)
    return cost
