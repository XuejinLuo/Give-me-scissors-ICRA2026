def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector is firmly gripping the keypoint of the scissors (keypoint 2)."""
    scissors_keypoint = keypoints[2]
    cost = np.linalg.norm(end_effector_1 - scissors_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Robot 2's end-effector is firmly gripping the keypoint of the hemostat (keypoint 1)."""
    hemostat_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_2 - hemostat_keypoint)
    return cost
