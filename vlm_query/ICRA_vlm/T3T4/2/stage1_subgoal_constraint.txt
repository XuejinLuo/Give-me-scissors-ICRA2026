def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector is firmly gripping the keypoint of the scissors."""
    scissors_keypoint = keypoints[3]  # Grasping the scissors at keypoint 3
    cost = np.linalg.norm(end_effector_1 - scissors_keypoint)
    return cost
