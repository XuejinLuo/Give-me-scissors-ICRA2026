def stage2_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector holds the scissors above 10 cm from the desk."""
    scissors_keypoint = keypoints[0]
    offsetted_point = scissors_keypoint + np.array([0, 0, 0.1])
    cost = np.linalg.norm(end_effector_1 - offsetted_point)
    return cost
