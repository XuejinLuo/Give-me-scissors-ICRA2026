# 1. Robot 1 picks up the scissors while Robot 2 picks up the hemostat.
# 2. Robot 1 holds the scissors above 10 cm and Robot 2 holds the hemostat above 10 cm.
# 3. Robot 1 releases the scissors on the desk and Robot 2 releases the hemostat on the desk.

### stage 1 sub-goal constraints
def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector is firmly gripping the keypoint of the scissors."""
    scissors_keypoint = keypoints[0]
    cost = np.linalg.norm(end_effector_1 - scissors_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):

    cost = np.linalg.norm(end_effector_1 - offsetted_point)
