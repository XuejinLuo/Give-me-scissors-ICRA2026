def stage3_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure that Robot 1's end-effector has released the scalpel on the desk at its original position."""
    scalpel_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - scalpel_keypoint)
    return cost
