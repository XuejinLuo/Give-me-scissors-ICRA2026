def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector is firmly gripping the scalpel's keypoint."""
    scalpel_keypoint = keypoints[2]
    cost = np.linalg.norm(end_effector_1 - scalpel_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Robot 2's end-effector is firmly gripping the tweezers' keypoint."""
    tweezers_keypoint = keypoints[4]
    cost = np.linalg.norm(end_effector_2 - tweezers_keypoint)
    return cost
