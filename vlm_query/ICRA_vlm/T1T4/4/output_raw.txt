# Stage 1: Robot 1 picks up the scalpel and Robot 2 picks up the hemostat.
# Stage 2: Robot 1 holds scalpel above 10 cm and Robot 2 holds hemostat above 10 cm.
# Stage 3: Robot 1 releases the scalpel on the desk and Robot 2 releases the hemostat on the desk.

### stage 1 sub-goal constraints
def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector is firmly gripping the keypoint of the scalpel."""
    scalpel_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - scalpel_keypoint)
    return cost

    cost = np.linalg.norm(end_effector_1 - offsetted_point)
