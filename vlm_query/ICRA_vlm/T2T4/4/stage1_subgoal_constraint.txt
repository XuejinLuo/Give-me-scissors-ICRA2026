def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector is firmly gripping the tweezers at keypoint 1."""
    grasp_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - grasp_keypoint)
    return cost
