def stage3_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector releases the tweezers on the desk."""
    release_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - release_keypoint)
    return cost
