def stage2_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector holds the tweezers 0.1m above the desk."""
    grasp_keypoint = keypoints[1]
    offsetted_point = grasp_keypoint + np.array([0, 0, 0.1])
    cost = np.linalg.norm(end_effector_1 - offsetted_point)
    return cost
