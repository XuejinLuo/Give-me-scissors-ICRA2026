def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector is firmly gripping the keypoint of the tweezers."""
    tweezers_keypoint = keypoints[2]  # Assuming keypoint 2 is for tweezers
    cost = np.linalg.norm(end_effector_1 - tweezers_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Robot 2's end-effector is firmly gripping the keypoint of the hemostat."""
    hemostat_keypoint = keypoints[1]  # Assuming keypoint 1 is for hemostat
    cost = np.linalg.norm(end_effector_2 - hemostat_keypoint)
    return cost
