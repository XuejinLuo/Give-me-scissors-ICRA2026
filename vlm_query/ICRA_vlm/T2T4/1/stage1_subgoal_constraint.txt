def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector is firmly gripping the tweezers at keypoint index 3."""
    tweezers_keypoint = keypoints[3]
    cost = np.linalg.norm(end_effector_1 - tweezers_keypoint)
    return cost
