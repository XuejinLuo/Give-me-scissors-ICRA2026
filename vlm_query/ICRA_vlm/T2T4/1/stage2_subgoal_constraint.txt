def stage2_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1 lifts the tweezers 10 cm above the desk."""
    tweezers_keypoint = keypoints[3]
    target_position = tweezers_keypoint + np.array([0, 0, 0.1])
    cost = np.linalg.norm(end_effector_1 - target_position)
    return cost
