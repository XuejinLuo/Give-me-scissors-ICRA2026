def stage3_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1 releases the tweezers on the desk. The cost is related to the end-effector's proximity to the desk's surface."""
    tweezers_keypoint = keypoints[3]
    desk_surface = tweezers_keypoint.copy()
    cost = np.linalg.norm(end_effector_1 - desk_surface)
    return cost
