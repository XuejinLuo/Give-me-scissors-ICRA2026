def stage3_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector releases the scalpel on the desk."""
    scalpel_keypoint = keypoints[3]
    cost = np.linalg.norm(end_effector_1 - scalpel_keypoint)
    return cost
