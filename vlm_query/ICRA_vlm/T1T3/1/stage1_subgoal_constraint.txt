def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector is grasping the scalpel at keypoint 3."""
    scalpel_keypoint = keypoints[3]
    cost = np.linalg.norm(end_effector_1 - scalpel_keypoint)
    return cost

def stage1_subgoal_constraint_robot2(end_effector_2, keypoints):

    scissors_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_2 - scissors_keypoint)
    return cost
