def stage1_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Ensure Robot 1's end-effector is firmly gripping the scalpel at keypoint 1."""
    scalpel_keypoint = keypoints[1]
    cost = np.linalg.norm(end_effector_1 - scalpel_keypoint)
    return cost
