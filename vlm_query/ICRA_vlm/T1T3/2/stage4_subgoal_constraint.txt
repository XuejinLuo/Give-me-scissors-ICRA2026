def stage4_subgoal_constraint_robot1(end_effector_1, keypoints):
    """Robot 1's end-effector maintains the position of the scalpel."""
    cost = np.linalg.norm(end_effector_1 - keypoints[3])
    return cost

def stage4_subgoal_constraint_robot2(end_effector_2, keypoints):
    """Robot 2's end-effector places the scissors back on the desk and releases it."""
    cost = np.linalg.norm(end_effector_2 - keypoints[1])
    return cost
